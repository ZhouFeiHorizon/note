   ```js
   /**
    * @param {string} s
    * @return {boolean}
    */
   var isValid = function (s) {

     const flagMap = {
      '(': ')',
       '[': ']',
       '{': '}'
     }

    //
     const array = []

     for (const cart of s) {
      if (flagMap[cart]) {
        // 右边
        array.push(flagMap[cart])
      }
      else  if (array.pop() !== cart)  {
        return false
      }
     }

    return array.length === 0
     // 你的代码
   }
   console.log(isValid("([])")) // true
   console.log(isValid("()[]({})")) // true
   console.log(isValid("([)]")) // false
   ```
