实现一个EventBus 时间20分钟

```js

// 实现一个EventBus 时间20分钟
class EventBus {
  on(){}
  once(){}
  emit(){}
  off(){}
  clear(){}
}

const EventBusIns = new EventBus();

function testCallback(name) {
 console.log(`${name} 出发了`);
}

const persons = ['小明', '小张', '小李']

EventBusIns.on('off-work', testCallback);	

EventBusIns.on('off-work', (name) => {
  console.log(`${name} 到家了`);
});

EventBusIns.once('notify', () => {
  console.log('下班了！');
});

EventBusIns.emit('notify');

persons.forEach(person => {
  EventBusIns.emit('off-work', person);
});

EventBusIns.emit('notify');

EventBusIns.off('off-work', testCallback);

persons.forEach(person => {
  EventBusIns.emit('off-work', person);
});

EventBusIns.clear();

persons.forEach(person => {
  EventBusIns.emit('off-work', person);
});

// 运行结果

// 下班了！
// 小明 出发了
// 小明 到家了
// 小张 出发了
// 小张 到家了
// 小李 出发了
// 小李 到家了

// 小明 到家了
// 小张 到家了
// 小李 到家了
```
